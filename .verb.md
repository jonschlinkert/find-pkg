Uses [find-file-up][], which is based on [look-up][] but faster since it does not support globs.

## Usage

```js
var findPkg = require('{%= name %}');

findPkg('.', function(err, fp) {
  //=> /User/jonschlinkert/dev/find-pkg/package.json
});

var fp = findPkg.sync('.');
//=> /User/jonschlinkert/dev/find-pkg/package.json
```
